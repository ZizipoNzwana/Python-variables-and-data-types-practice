2ï¸âƒ£ Data Manipulation
Task: Dictionary Aggregation

Goal:
Create a function count_word_occurrences(words) that:

Accepts a list of strings

Returns a dictionary where keys are words and values are occurrence counts

# test_data_manipulation.py
import pytest

def test_counts_words_correctly():
    words = ["apple", "banana", "apple", "orange"]
    result = count_word_occurrences(words)
    assert result == {
        "apple": 2,
        "banana": 1,
        "orange": 1
    }

def test_empty_input():
    result = count_word_occurrences([])
    assert result == {}

def test_case_sensitive():
    words = ["Apple", "apple"]
    result = count_word_occurrences(words)
    assert result == {
        "Apple": 1,
        "apple": 1
    }
